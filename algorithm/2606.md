typedef struct { 
    int32_t** graph;
    int32_t* neighbors;
    int32_t* count;
    int32_t* visited;
    int32_t* N;
} Graph;

int32_t dfs(Graph* a, int32_t R) {
    if ((*a).visited[R]) {
        return 0;
    }
	if (R != 1) {
		*(*a).count += 1;
	}

    *((*a).visited + R) = 1;

    for (int32_t i = 0; i < (*a).neighbors[R]; i += 1) {
        if (!(*a).visited[a->graph[R][i]]) {
            dfs(a, a->graph[R][i]);
        }
    }

    return 0;
}
